15/////////////////////////////////////////////////////////////////////////////
16/// Filesystem initialization macro.
17/// This macro must be instantiated before using nodel::filesystem services.
18/// It defines thread-local data-structures.
19/////////////////////////////////////////////////////////////////////////////
20#define NODEL_INIT_FILESYSTEM namespace nodel::filesystem::impl { thread_local Registry default_registry; }
21extern thread_local Registry default_registry;
22
23inline
24void init_default_registry() {
25 default_registry.set_directory_default<SubDirectory>();
26 default_registry.set_file_default<GenericFile>();
27 default_registry.associate<JsonFile>(".json");
28 default_registry.associate<CsvFile>(".csv");
29// default_registry.associate<ZipFile>(".zip");
30 default_registry.associate<GenericFile>(".txt");
31}
32
33} // namespace impl
34
35/////////////////////////////////////////////////////////////////////////////
36/// Returns the *thread-local* default registry.
37/// Changes to the file associations in this registry will affect *all*
38/// Objects bound to the filesystem in the current thread.
39/////////////////////////////////////////////////////////////////////////////
40inline
41Registry& default_registry() {
42 if (impl::default_registry.is_empty()) {
43 impl::init_default_registry();
44 }
45 return impl::default_registry;
46}

48/////////////////////////////////////////////////////////////////////////////
49/// Configure Nodel filesystem.
50/// - Enable binding URI with the "file" scheme using
51/// nodel::bind(const String& uri_spec, ...).
52/// - By default, Objects bound to the filesystem have their DataSource mode
53/// set to read-only. See @ref DataSource::Options for more information.
54/// - Use URI query to set path when the path is relative
55/// (ex: file://?path=a/b).
56/////////////////////////////////////////////////////////////////////////////
57inline
58void configure(DataSource::Options default_options={}) {
59 register_uri_scheme("file", [default_options] (const Object& uri, DataSource::Origin origin) -> DataSource* {
60 auto p_ds = new Directory(new Registry{default_registry()}, origin);
61 p_ds->set_options(default_options);
62 return p_ds;
63 });
64}
