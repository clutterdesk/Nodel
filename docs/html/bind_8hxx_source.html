33Object bind(const URI& uri, const Object& obj) {

35 auto scheme = uri.get("scheme"_key).as<String>();
36 auto factory = lookup_uri_scheme(scheme);
37 if (factory) {
38 auto origin = (obj == nil)? DataSource::Origin::SOURCE: DataSource::Origin::MEMORY;
39 auto p_ds = factory(uri, origin);
40 if (obj == nil) bound = p_ds; else p_ds->bind(bound);

43 } else {
44 std::stringstream ss;
