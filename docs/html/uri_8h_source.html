47Object URI::parse_uri_query(const String& query) {
48 Object result = Object::OMAP;
49 auto it = query.cbegin();
50 auto end = query.cend();
51 std::string key;
52 std::string val;
53 std::string* p_target = &key;
54 for (; it != end; ++it) {
55 char c = *it;
56 if (c == '=') {
57 p_target = &val;
58 } else if (c == '&' || c == ';') {
59 p_target = &key;
60 result.set(key, val);
61 key.clear();
62 val.clear();
63 } else {
64 p_target->push_back(c);
65 }
66 }
67 if (key.size() != 0)
68 result.set(key, val);
69 return result;
70}

72inline
73void URI::parse(const String& spec) {
74 static std::regex uri_re{R"(([^:]+)://(([^@]+\@)?([^:/?#]*)(:(\d+))?)?((/[^?#]*)?(\?[^#]+)?(\#.*)?)?)"};
75
76 std::smatch match;
77 if (std::regex_match(spec, match, uri_re)) {
78 (*(Object*)this) = Object{Object::OMAP};
79 if (match[1].length() > 0) set("scheme"_key, match[1].str());
80 if (match[3].length() > 0) set("user"_key, substr(match[3].str(), 0, -1));
81 if (match[4].length() > 0) set("host"_key, match[4].str());
82 if (match[6].length() > 0) set("port"_key, Object(match[6].str()).cast<Int>());
83 if (match[8].length() > 0) set("path"_key, match[8].str());
84 if (match[9].length() > 0) set("query"_key, parse_uri_query(match[9].str().substr(1)));
85 if (match[10].length() > 0) set("fragment"_key, match[10].str().substr(1));
86 }

103inline
104DataSourceFactory lookup_uri_scheme(const String& scheme) {
105 if (local_uri_map.size() == 0) {
106 std::scoped_lock lock{global_uri_map_mutex};
107 local_uri_map = global_uri_map;
108 if (auto it = local_uri_map.find(scheme); it != local_uri_map.end())
109 return std::get<1>(*it);
110 } else {
111 // If scheme is not found in local_uri_map, it might have been registered after the first
112 // lookup for this thread. In that case, we check the global_uri_map, with locking, and
113 // then update the local_uri_map.
114 auto it = local_uri_map.find(scheme);
115 if (it != local_uri_map.end()) {
116 return std::get<1>(*it);
117 } else {
118 std::scoped_lock lock{global_uri_map_mutex};
119 it = global_uri_map.find(scheme);
120 if (it != global_uri_map.end()) {
121 local_uri_map = global_uri_map;
122 return std::get<1>(*it);
123 }
124 }
125 }
126 return {};
127}
