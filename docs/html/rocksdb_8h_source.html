26/////////////////////////////////////////////////////////////////////////////
27/// Enable and configure the URI "rocksdb" scheme.
28/// - Enable binding URI with the "rocksdb" scheme using nodel::bind(const String& uri_spec, ...).
29/// - Create directory association for ".rocksdb" in the default registry.
30/// @see nodel::bind(const String&, Object)
31/////////////////////////////////////////////////////////////////////////////
32inline
33void configure(Options options={}) {
34 register_uri_scheme("rocksdb", [options] (const URI& uri, DataSource::Origin origin) -> DataSource* {
35 auto p_ds = new rocksdb::DB(origin);
36 p_ds->set_options(options.base);
37 p_ds->set_db_options(options.db);
38 p_ds->set_read_options(options.db_read);
39 p_ds->set_write_options(options.db_write);
40 return p_ds;
41 });
42
43 filesystem::default_registry().associate<rocksdb::DB>(options.db_ext);
44}

46/////////////////////////////////////////////////////////////////////////////
47/// Register a directory extension to recognize RocksDB database directories.
48/// This function can be used to customize the extension used in a single tree, when that
49/// extension is different from the default extension.
50/// @param fs_obj A filesystem object.
51/// @param ext The directory extension, including the leading period.
52/// @note The extension is registered for the entire tree.
53/////////////////////////////////////////////////////////////////////////////
54inline
55void register_directory_extension(const Object& fs_obj, const std::string& ext = ".rocksdb") {
56 Ref<Registry> r_reg = filesystem::get_registry(fs_obj);
57 if (!r_reg) {
58 WARN("Argument must be a filesystem directory object.");
59 return;
60 }
61 r_reg->associate<rocksdb::DB>(ext);
62}
